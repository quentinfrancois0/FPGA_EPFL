#include <inttypes.h>
#include "system.h"
#include "io.h"
#include "altera_avalon_pio_regs.h"
#include <unistd.h>

void InitPort(void) {
	IOWR_8DIRECT(PARALLEL_PORT0_0_BASE, 0x00, 0xff); //PWM clock divider = 0xffff => frequency = 3 Hz
	IOWR_8DIRECT(PARALLEL_PORT0_0_BASE, 0x01, 0xff);
	IOWR_8DIRECT(PARALLEL_PORT0_0_BASE, 0x02, 0x80); //PWM duty cycle = 50%
	IOWR_8DIRECT(PARALLEL_PORT0_0_BASE, 0x03, 0x01); //PWM polarity = 1
	IOWR_8DIRECT(PARALLEL_PORT0_0_BASE, 0x05, 0x01); //PWM output control = 1
}

int main(void) {

	InitPort();

	while(1) {} //infinite loop

	return 1;
}
